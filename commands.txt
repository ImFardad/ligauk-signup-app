# ===============================================
# دستورات ضروری برای مدیریت پروژه (Docker و PM2)
# ===============================================


# --- مدیریت با Docker Compose (روش پیشنهادی برای توسعه) ---

# اجرا کردن کامل پروژه (اپلیکیشن + دیتابیس) و نمایش لاگ‌ها
docker-compose up

# اجرا کردن پروژه و ساخت مجدد ایمیج‌ها (وقتی Dockerfile را تغییر می‌دهید)
docker-compose up --build

# اجرا کردن پروژه در پس‌زمینه (ترمینال آزاد می‌شود)
docker-compose up -d

# متوقف کردن و حذف کامل کانتینرهای پروژه
docker-compose down

# مشاهده لاگ‌های زنده اپلیکیشن
docker-compose logs -f app


# --- مدیریت با PM2 (برای اجرا مستقیم روی سرور بدون داکر) ---

# اجرا کردن اپلیکیشن در حالت کلاستر (استفاده از تمام هسته‌های CPU)
pm2 start app.js -i max --name "my-app"

# متوقف کردن اپلیکیشن
pm2 stop my-app

# ری‌استارت کردن اپلیکیشن
pm2 restart my-app

# مشاهده لیست تمام پروسه‌های در حال اجرا
pm2 list

# مشاهده لاگ‌های زنده
pm2 logs

# حذف کامل اپلیکیشن از لیست PM2
pm2 delete my-app


# --- دستورات عمومی NPM ---

# نصب تمام پکیج‌ها از package.json
npm install

# نصب یک پکیج جدید
npm install <package_name>

# حذف یک پکیج
npm uninstall <package_name>
